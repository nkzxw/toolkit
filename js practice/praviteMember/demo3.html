Patterns

Public

    function Constructor(...){
        this.membername = value;
    }
    Constructor.prototype.membername = value;

Private

    function Constructor(...){
        var that = this;
        var membername = value;

        function member(...){...}
    }

    Note: The function statement
        function member(...){...}
    is shorthand for
        var member = function(...){...}

Privileged

    function Constructor(...){
        this.member = function(...){...}
    }
